    /***********************

    const STORAGE_KEY = "qna_admin_demo_v1";
    /** @type {{id:string, question:string, answer:string, keywords:string[], updatedAt:number}[]} */
    let qas = [];
    let searchTerm = "";
    let drawerMode = "create"; // create | edit
    let editingId = null;
    let draftTags = [];

    const els = {
      tbody: document.getElementById("tbody"),
      empty: document.getElementById("empty"),
      countMeta: document.getElementById("countMeta"),
      updatedMeta: document.getElementById("updatedMeta"),

      searchInput: document.getElementById("searchInput"),
      addBtn: document.getElementById("addBtn"),

      overlay: document.getElementById("overlay"),
      drawer: document.getElementById("drawer"),
      drawerTitle: document.getElementById("drawerTitle"),
      closeDrawerBtn: document.getElementById("closeDrawerBtn"),

      qInput: document.getElementById("qInput"),
      aInput: document.getElementById("aInput"),

      qError: document.getElementById("qError"),
      aError: document.getElementById("aError"),

      tagWrap: document.getElementById("tagWrap"),
      tagInput: document.getElementById("tagInput"),

      cancelBtn: document.getElementById("cancelBtn"),
      saveBtn: document.getElementById("saveBtn"),

      toast: document.getElementById("toast"),
    };

    function now() { return Date.now(); }
    function pad2(n){ return String(n).padStart(2,"0"); }
    function formatDate(ts){
      const d = new Date(ts);
      return `${pad2(d.getMonth()+1)}/${pad2(d.getDate())} ${pad2(d.getHours())}:${pad2(d.getMinutes())}`;
    }
    function uid(){
      return Math.random().toString(16).slice(2) + "-" + Math.random().toString(16).slice(2);
    }

    function load(){
      try{
        const raw = localStorage.getItem(STORAGE_KEY);
        if(raw){
          qas = JSON.parse(raw);
          return;
        }
      }catch(e){}
      // seed data
      qas = [
        {
          id: uid(),
          question: "미니맵은 어디서 켜나요?",
          answer: "미니맵은 화면 우측 상단 UI에서 활성화할 수 있습니다. 설정 > UI 메뉴에서 위치/크기도 조절 가능합니다.",
          keywords: ["미니맵", "UI", "설정"],
          updatedAt: now() - 1000*60*60*6
        },
        {
          id: uid(),
          question: "아이템을 버리려면 어떻게 해요?",
          answer: "인벤토리에서 아이템을 선택한 뒤 ‘버리기’를 누르면 됩니다. 일부 아이템은 버리기가 제한될 수 있습니다.",
          keywords: ["아이템", "인벤토리"],
          updatedAt: now() - 1000*60*35
        }
      ];
      persist();
    }

    function persist(){
      try{
        localStorage.setItem(STORAGE_KEY, JSON.stringify(qas));
      }catch(e){}
      render();
    }

    /***********************
     * Rendering
     ***********************/
    function getFiltered(){
      const t = searchTerm.trim().toLowerCase();
      if(!t) return qas.slice().sort((a,b)=>b.updatedAt-a.updatedAt);
      return qas
        .filter(item => {
          const inQ = item.question.toLowerCase().includes(t);
          const inK = item.keywords.join(" ").toLowerCase().includes(t);
          return inQ || inK;
        })
        .sort((a,b)=>b.updatedAt-a.updatedAt);
    }

    function render(){
      const list = getFiltered();
      els.tbody.innerHTML = "";

      // meta
      els.countMeta.textContent = `총 ${list.length}건`;
      const latest = qas.length ? Math.max(...qas.map(x=>x.updatedAt)) : null;
      els.updatedMeta.textContent = `마지막 업데이트: ${latest ? formatDate(latest) : "-"}`;

      if(list.length === 0){
        els.empty.style.display = "block";
      }else{
        els.empty.style.display = "none";
      }

      for(const item of list){
        const tr = document.createElement("tr");

        const tdQ = document.createElement("td");
        tdQ.className = "col-question";
        tdQ.textContent = item.question;

        const tdK = document.createElement("td");
        tdK.className = "col-keywords";
        const chips = document.createElement("div");
        chips.className = "chips";
        (item.keywords || []).slice(0, 6).forEach(k=>{
          const c = document.createElement("span");
          c.className = "chip";
          c.textContent = k;
          chips.appendChild(c);
        });
        if((item.keywords || []).length > 6){
          const more = document.createElement("span");
          more.className = "chip";
          more.textContent = `+${item.keywords.length - 6}`;
          chips.appendChild(more);
        }
        tdK.appendChild(chips);

        const tdU = document.createElement("td");
        tdU.className = "col-updated";
        tdU.textContent = formatDate(item.updatedAt);

        const tdA = document.createElement("td");
        tdA.className = "col-actions";
        const actions = document.createElement("div");
        actions.className = "actions";

        const editBtn = document.createElement("button");
        editBtn.className = "btn btn-mini";
        editBtn.type = "button";
        editBtn.textContent = "수정";
        editBtn.addEventListener("click", (e)=>{
          e.stopPropagation();
          openDrawer("edit", item.id);
        });

        const delBtn = document.createElement("button");
        delBtn.className = "btn btn-mini btn-danger";
        delBtn.type = "button";
        delBtn.textContent = "삭제";
        delBtn.addEventListener("click", (e)=>{
          e.stopPropagation();
          removeItem(item.id);
        });

        actions.appendChild(editBtn);
        actions.appendChild(delBtn);
        tdA.appendChild(actions);

        // row click also opens edit (optional)
        tr.addEventListener("click", ()=>{
          openDrawer("edit", item.id);
        });

        tr.appendChild(tdQ);
        tr.appendChild(tdK);
        tr.appendChild(tdU);
        tr.appendChild(tdA);
        els.tbody.appendChild(tr);
      }
    }

    /***********************
     * Drawer
     ***********************/
    function setDrawerOpen(open){
      if(open){
        els.overlay.classList.add("open");
        els.drawer.classList.add("open");
        els.overlay.setAttribute("aria-hidden","false");
        // focus first input
        setTimeout(()=>els.qInput.focus(), 0);
      }else{
        els.overlay.classList.remove("open");
        els.drawer.classList.remove("open");
        els.overlay.setAttribute("aria-hidden","true");
        clearErrors();
      }
    }

    function clearForm(){
      els.qInput.value = "";
      els.aInput.value = "";
      draftTags = [];
      renderTagChips();
      autoGrow(els.aInput);
      clearErrors();
    }

    function clearErrors(){
      els.qError.classList.remove("show");
      els.aError.classList.remove("show");
    }

    function openDrawer(mode, id=null){
      drawerMode = mode;
      editingId = id;

      if(mode === "create"){
        els.drawerTitle.textContent = "신규 Q&A 등록";
        clearForm();
      }else{
        els.drawerTitle.textContent = "Q&A 수정";
        const item = qas.find(x=>x.id===id);
        if(!item) return;
        els.qInput.value = item.question || "";
        els.aInput.value = item.answer || "";
        draftTags = Array.isArray(item.keywords) ? [...item.keywords] : [];
        renderTagChips();
        autoGrow(els.aInput);
        clearErrors();
      }

      setDrawerOpen(true);
    }

    function closeDrawer(){
      setDrawerOpen(false);
      editingId = null;
      drawerMode = "create";
    }

    /***********************
     * Tags (Keyword input)
     ***********************/
    function normalizeTag(s){
      return (s || "")
        .trim()
        .replace(/\s+/g," ");
    }

    function addTag(raw){
      const t = normalizeTag(raw);
      if(!t) return;
      if(draftTags.some(x=>x.toLowerCase()===t.toLowerCase())) return; // dedupe
      draftTags.push(t);
      renderTagChips();
    }

    function removeTag(index){
      draftTags.splice(index,1);
      renderTagChips();
    }

    function renderTagChips(){
      // remove all tag chips but keep input at the end
      const inputEl = els.tagInput;
      // clear wrap
      els.tagWrap.innerHTML = "";
      // add chips
      draftTags.forEach((t, idx)=>{
        const chip = document.createElement("span");
        chip.className = "tag-chip";
        chip.title = t;

        const text = document.createElement("span");
        text.textContent = t;

        const btn = document.createElement("button");
        btn.type = "button";
        btn.setAttribute("aria-label", `키워드 삭제: ${t}`);
        btn.textContent = "×";
        btn.addEventListener("click", (e)=>{
          e.stopPropagation();
          removeTag(idx);
          inputEl.focus();
        });

        chip.appendChild(text);
        chip.appendChild(btn);
        els.tagWrap.appendChild(chip);
      });

      // re-append input
      els.tagWrap.appendChild(inputEl);
      inputEl.value = "";
    }

    /***********************
     * Save / Delete
     ***********************/
    function validate(){
      clearErrors();
      const q = els.qInput.value.trim();
      const a = els.aInput.value.trim();
      let ok = true;
      if(!q){
        els.qError.classList.add("show");
        ok = false;
      }
      if(!a){
        els.aError.classList.add("show");
        ok = false;
      }
      return ok;
    }

    function upsert(){
      if(!validate()) return;

      const payload = {
        question: els.qInput.value.trim(),
        answer: els.aInput.value.trim(),
        keywords: [...draftTags],
        updatedAt: now(),
      };

      if(drawerMode === "create"){
        qas.push({ id: uid(), ...payload });
        persist();
        showToast("저장되었습니다.");
        closeDrawer();
      }else{
        const idx = qas.findIndex(x=>x.id===editingId);
        if(idx === -1) return;
        qas[idx] = { ...qas[idx], ...payload };
        persist();
        showToast("수정되었습니다.");
        closeDrawer();
      }
    }

    function removeItem(id){
      const item = qas.find(x=>x.id===id);
      if(!item) return;
      const ok = confirm(`삭제할까요?\n\n질문: ${item.question}`);
      if(!ok) return;
      qas = qas.filter(x=>x.id!==id);
      persist();
      showToast("삭제되었습니다.");
    }

    /***********************
     * Textarea auto-grow
     ***********************/
    function autoGrow(textarea){
      textarea.style.height = "auto";
      // cap at ~260px then internal scroll
      const max = 260;
      const h = Math.min(textarea.scrollHeight, max);
      textarea.style.height = h + "px";
      textarea.style.overflowY = textarea.scrollHeight > max ? "auto" : "hidden";
    }

    /***********************
     * Toast
     ***********************/
    let toastTimer = null;
    function showToast(msg){
      els.toast.textContent = msg;
      els.toast.classList.add("show");
      clearTimeout(toastTimer);
      toastTimer = setTimeout(()=>els.toast.classList.remove("show"), 1600);
    }

    /***********************
     * Events
     ***********************/
    els.searchInput.addEventListener("input", (e)=>{
      searchTerm = e.target.value || "";
      render();
    });

    els.addBtn.addEventListener("click", ()=>{
      openDrawer("create");
    });

    els.closeDrawerBtn.addEventListener("click", closeDrawer);
    els.cancelBtn.addEventListener("click", closeDrawer);

    els.overlay.addEventListener("click", closeDrawer);

    document.addEventListener("keydown", (e)=>{
      if(e.key === "Escape"){
        // close drawer if open
        if(els.drawer.classList.contains("open")) closeDrawer();
      }
    });

    els.saveBtn.addEventListener("click", upsert);

    // Tag input: Enter or comma
    els.tagInput.addEventListener("keydown", (e)=>{
      const v = els.tagInput.value;

      if(e.key === "Enter"){
        e.preventDefault();
        addTag(v);
        return;
      }
      if(e.key === "," || e.key === "Comma"){
        e.preventDefault();
        addTag(v.replace(/,/g,""));
        return;
      }
      if(e.key === "Backspace" && !v){
        // remove last tag
        if(draftTags.length){
          draftTags.pop();
          renderTagChips();
        }
      }
    });

    // Also handle paste with commas
    els.tagInput.addEventListener("paste", (e)=>{
      const text = (e.clipboardData || window.clipboardData).getData("text");
      if(text && text.includes(",")){
        e.preventDefault();
        text.split(",").map(s=>s.trim()).filter(Boolean).forEach(addTag);
      }
    });

    // Textarea auto-grow
    els.aInput.addEventListener("input", ()=>autoGrow(els.aInput));

    // initial
    load();
    render();
    autoGrow(els.aInput);
